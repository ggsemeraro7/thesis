##DEGENERATION CODE by FASTPCR
Degenerate codes: B=CGT, D=AGT, H=ACT, K=GT, M=AC, N=ACGT, R=AG, S=GC, V=ACG, W=AT, Y=CT, U=Uracil, I=Inosine.


##PRIMER DEGENERATO nelle sue diverse versioni per NCBI BLAST

primer="CAATCAAAGCCCAGAGCAGAGCCCCARTCRTG"
prefix="linb2ar_"

declare -A degenerate=(
  [B]="C G T"
  [D]="A G T"
  [H]="A C T"
  [K]="G T"
  [M]="A C"
  [N]="A C G T"
  [R]="A G"
  [S]="G C"
  [V]="A C G"
  [W]="A T"
  [Y]="C T"
  [U]="U"   # uracil (non espande)
  [I]="I"   # inosina (non espande)
)

positions=()
for ((i=0; i<${#primer}; i++)); do
  base="${primer:$i:1}"
  if [[ -n "${degenerate[$base]}" && "${degenerate[$base]}" != "$base" ]]; then
    positions+=("$i")
  fi
done

expand() {
  local seq="$1"
  local idx="$2"
  local tag="$3"

  if (( idx == ${#positions[@]} )); then
    echo ">${prefix}${tag}"
    echo "$seq"
    return
  fi

  local pos="${positions[$idx]}"
  local code="${seq:$pos:1}"

  for b in ${degenerate[$code]}; do
    local newseq="${seq:0:$pos}${b}${seq:$((pos+1))}"
    expand "$newseq" $((idx+1)) "${tag}${b}"
  done
}

expand "$primer" 0 ""

import re

input_file = "tcbq3_mg_fastpcr_mis2.txt"
output_fasta = "tcbq3_mg_fastpcr_with_fullID.fasta"

with open(input_file) as f:
    lines = f.readlines()

current_seq_full = None
primer_f = None
pos_f = None
primer_r = None
pos_r = None
amplicon_len = None
amplicon_seq = []
in_amplicon = False

with open(output_fasta, "w") as out:

    for line in lines:
        line = line.rstrip()

        # --- Nuova sezione target: cattura tutta la descrizione dopo "search for:" ---
        m = re.match(r"In silico Primer\(s\) search for:\s+(.+)", line)
        if m:
            # prima, salva l’eventuale amplicone precedente
            if in_amplicon and amplicon_seq:
                header = (
                    f">{current_seq_full}|{primer_f}:{pos_f}|{primer_r}:{pos_r}|{amplicon_len}bp"
                )
                out.write(header + "\n")
                out.write("".join(amplicon_seq) + "\n")

            # ora resetto per la nuova sezione
            current_seq_full = m.group(1).strip()
            primer_f = None
            pos_f = None
            primer_r = None
            pos_r = None
            amplicon_len = None
            amplicon_seq = []
            in_amplicon = False
            continue

        # --- Binding site and primer name, forward ---
        m = re.match(r">(\S+)\s+(\d+)->(\d+)", line)
        if m:
            # se forward non ancora salvato, salvalo
            if primer_f is None:
                primer_f = m.group(1)
                pos_f = f"{m.group(2)}-{m.group(3)}"
            # se invece era già presente, ignora qui
            continue

        # --- Binding site and primer name, reverse ---
        m = re.match(r">(\S+)\s+(\d+)<-(\d+)", line)
        if m:
            primer_r = m.group(1)
            pos_r = f"{m.group(3)}-{m.group(2)}"
            continue

        # --- Amplicon header: ottengo lunghezza prodotto ---
        m = re.match(r">(\d+)-(\d+)\s+Amplicon size:\s+(\d+)bp", line)
        if m:
            amplicon_len = m.group(3)
            amplicon_seq = []
            in_amplicon = True
            continue

        # --- Amplicon sequence lines ---
        if in_amplicon:
            if line == "" or line.startswith("___"):
                # fine di un blocco amplicone
                if amplicon_seq:
                    header = (
                        f">{current_seq_full}|{primer_f}:{pos_f}|{primer_r}:{pos_r}|{amplicon_len}bp"
                    )
                    out.write(header + "\n")
                    out.write("".join(amplicon_seq) + "\n")

                amplicon_seq = []
                in_amplicon = False
            else:
                amplicon_seq.append(line)

    # --- Scrivi l’ultimo amplicone se fine file ---
    if in_amplicon and amplicon_seq:
        header = (
            f">{current_seq_full}|{primer_f}:{pos_f}|{primer_r}:{pos_r}|{amplicon_len}bp"
        )
        out.write(header + "\n")
        out.write("".join(amplicon_seq) + "\n")

import csv
import os
from datetime import date

fasta_file = "linb_mg_fastpcr_mis2_amplicon.fasta"
output_file = "linb_amplicons_info.tsv"

# === INFO ANALISI ===
analysis_id = "run_01"
sample = "S01"
status = "planned"
notes = ""
analysis_date = date.today().strftime("%d/%m/%Y")

# === determina il prossimo mis ===
mis_number = 1

if os.path.exists(output_file):
    with open(output_file, "r") as f:
        rows = [line for line in f if line.strip()]
        if len(rows) > 1:  # header + almeno una riga
            last_line = rows[-1].split("\t")
            last_mis = last_line[5]          # colonna "mis"
            mis_number = int(last_mis.split("_")[1]) + 1

mis = f"mis_{mis_number}"

file_exists = os.path.exists(output_file)

with open(fasta_file) as f, open(output_file, "a", newline="") as out:
    writer = csv.writer(out, delimiter="\t")

    # header SOLO se il file non esiste
    if not file_exists:
        writer.writerow([
            "seqID", "fwd", "rev", "length(bp)",
            "analysis_id", "mis", "sample", "date", "status", "notes"
        ])

    for line in f:
        if line.startswith(">"):
            header = line.strip()[1:]
            seqid, primers, length = header.split("|")

            fwd, rev = primers.split("-")
            length_bp = length.replace("bp", "")

            writer.writerow([
                seqid, fwd, rev, length_bp,
                analysis_id, mis, sample, analysis_date, status, notes
            ])

print(f"Done! Added analysis with {mis}")
