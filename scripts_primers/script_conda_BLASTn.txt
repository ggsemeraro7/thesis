##SCRIPT CONDA BLAST N
#Crea database 
makeblastdb -dbtype nucl -in   uniprot_EC_3.8.1.5_linb_nt.fasta -out file_blastn_u_linb/db_kegg_nt_linb

#Crea tabella omologie
blastn -db file_blastn_u_linb/db_kegg_nt_linb -query  ITApCDS_EC_3.8.1.5_K01563_NT.fasta -outfmt 6 -word_size 11 -out blastn_uniprot_EC_3.8.1.5_linb_NT.tsv

#FILTRO SEQ MIGLIORE BLAST ED UNIVOCHE

##FILTRAGGIO SEQUENZE 
#COMANDO PER FILTRARE SOLO LA HIT CON E-VALUE PIU BASSO PER OGNI QUERY BLASTN/DIAMOND 
gawk -F'\t' '
NF<11 {next}  # salta righe strane/vuote
{
  q=$1
  ev=$11+0

  if (!(q in bestlen) || len>bestlen[q] || (len==bestlen[q] && ev<bestev[q])) {
    bestlen[q]=len
    bestev[q]=ev
    best[q]=$0
  }
}
END{
  for (q in best) print best[q]
}
' blastn_uniprot_EC_3.8.1.5_linb_NT.tsv > best_Evalue_blastn_uniprot_EC_3.8.1.5_linb_NT.tsv


#COMANDO PER FILTRARE SOLO LE SEQUENZE UNIVOCHE DELLA SEQUENZA TARGET
awk -F'\t' '
{
  key = $2
  val = $11+0  # convertiamo in numero
  if (!(key in min) || val < min[key]) {
    min[key] = val
    line[key] = $0
  }
}
END {
  for (k in line)
    print line[k]
}
' best_Evalue_blastn_uniprot_EC_3.8.1.5_linb_NT.tsv > unique_best_Evalue_blastn_uniprot_EC_3.8.1.5_linb_NT.tsv

##COMANDO PER ESTRARRE FULLCDS DA FILE EC OMNICOMPRENSIVO
#Crea un file con le full CDS da utilizzare per le analisi successive (fullCDS_LINB_nt.fasta)
cut -f2 unique_best_Evalue_blastn_uniprot_EC_3.8.1.5_linb_NT.tsv | sort -u > ids_unique_best_Evalue_blastn_uniprot_EC_3.8.1.5_linb_NT.tsv
seqkit grep -f  ids_unique_best_Evalue_blastn_uniprot_EC_3.8.1.5_linb_NT.tsv   uniprot_EC_3.8.1.5_linb_nt.fasta > ITAfCDS_uniprot_EC_3.8.1.5_linb_nt.fasta

TABELLA HEADERS
   1.  qseqid      query or source (gene) sequence id

   2.  sseqid      subject or target (reference genome) sequence id

   3.  pident      percentage of identical positions

   4.  length      alignment length (sequence overlap)

   5.  mismatch    number of mismatches

   6.  gapopen     number of gap openings

   7.  qstart      start of alignment in query

   8.  qend        end of alignment in query

   9.  sstart      start of alignment in subject

 10.  send        end of alignment in subject

 11.  evalue      expect value

 12.  bitscore    bit score
